--Problema 1
SELECT first_name, last_name, TO_CHAR(salary, '$999,999.0') AS sueldo
FROM employees;
--Problema 2
SELECT first_name, last_name,
UPPER(REPLACE(CONCAT(SUBSTR(first_name, 1, 1), last_name), ' ')) AS nombre_de_usuario
FROM employees;
--Problema 3
SELECT
CONCAT(CONCAT(first_name, ' '), last_name) AS empleado,
(salary + 1200) * 1.05 AS salario
FROM employees
WHERE first_name = 'John' AND last_name = 'Russell';
--Problema 4
SELECT first_name, last_name, 
TRUNC((SYSDATE - hire_date) / 365) AS anhos_antiguedad
FROM employees;
--Problema 5
SELECT employee_id, first_name, last_name, hire_date,
NEXT_DAY(ADD_MONTHS(hire_date, 6), 'LUNES') AS fecha_revision
FROM employees;
--Problema 6
SELECT first_name, last_name, TO_CHAR(salary, '$999,999') AS salario,
TO_CHAR(hire_date, 'DAY DD, MONTH "DE" YYYY') AS fecha_contratacion
FROM employees;
--Problema 7
SELECT * 
FROM employees
WHERE hire_date BETWEEN '01/07/00' AND '31/12/00';
